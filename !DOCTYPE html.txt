try{sess=JSON.parse(localStorage.getItem('hrk_v4')||'[]')}catch(e){}

// VOICE
let recog=null,micOn=false;
(function(){
  const SR=window.SpeechRecognition||window.webkitSpeechRecognition;
  if(!SR){const b=document.getElementById('mic-btn');if(b)b.style.display='none';return;}
  recog=new SR();recog.lang='ru-RU';recog.continuous=false;recog.interimResults=false;
  recog.onresult=e=>{const t=e.results[0][0].transcript;const i=document.getElementById('inp');i.value=(i.value+' '+t).trim();onInput();stopMic();};
  recog.onerror=recog.onend=()=>stopMic();
})();
function toggleMic(){micOn?stopMic():startMic();}
function startMic(){if(!recog)return;try{recog.start();micOn=true;document.getElementById('mic-btn').classList.add('on');}catch(e){}}
function stopMic(){if(!recog)return;try{recog.stop();}catch(e){}micOn=false;document.getElementById('mic-btn').classList.remove('on');}

function onInput(){
  const v=document.getElementById('inp').value;
  const c=document.getElementById('char-cnt');
  if(c)c.textContent=v.length+' / 500';
  if(v.length>500)document.getElementById('inp').value=v.slice(0,500);
}

function go(id){
  document.querySelectorAll('.screen').forEach(s=>s.classList.remove('on'));
  document.querySelectorAll('.nb').forEach(b=>b.classList.remove('on'));
  document.getElementById('s-'+id).classList.add('on');
  const b=document.querySelector(`.nb[onclick="go('${id}')"]`);if(b)b.classList.add('on');
  if(id==='profile')renderProfile();
}

function startSc(sc){
  cur=sc;hist=[];cnt=0;round=0;stopMic();
  document.getElementById('cb-title').textContent=sc.title;
  document.getElementById('cb-char').textContent=sc.char;
  document.getElementById('coach-text').textContent=sc.coach;
  document.getElementById('round-info').textContent='Раунд 0 / 5';
  document.getElementById('msgs').innerHTML='';
  document.getElementById('inp').value='';
  document.getElementById('char-cnt').textContent='0 / 500';
  go('chat');
  // Show character opening
  const d=document.getElementById('msgs');
  const el=document.createElement('div');el.className='char-msg';
  el.innerHTML=`<div class="char-name">${sc.char}</div><div class="char-action">${sc.action}</div><div class="char-quote">${sc.open}</div><div class="char-hint">${sc.hint}</div>`;
  d.appendChild(el);
}

function addMsg(role,text){
  const d=document.getElementById('msgs');
  const div=document.createElement('div');div.className='msg'+(role==='user'?' u':'');
  div.innerHTML=`<div class="ava">${role==='user'?'Вы':'ИИ'}</div><div class="bub">${text}</div>`;
  d.appendChild(div);d.scrollTop=d.scrollHeight;
  if(role==='user'){cnt++;round++;document.getElementById('round-info').textContent=`Раунд ${Math.min(round,5)} / 5`;}
}

function addTyping(){
  const d=document.getElementById('msgs');
  const el=document.createElement('div');el.id='typ';el.className='msg';
  el.innerHTML=`<div class="ava">ИИ</div><div class="bub"><div class="dots"><span></span><span></span><span></span></div></div>`;
  d.appendChild(el);d.scrollTop=d.scrollHeight;
}
function rmTyping(){const t=document.getElementById('typ');if(t)t.remove();}

async function send(){
  const inp=document.getElementById('inp');
  const text=inp.value.trim();if(!text||!cur)return;
  inp.value='';document.getElementById('char-cnt').textContent='0 / 500';
  document.getElementById('send-btn').disabled=true;
  addMsg('user',text);hist.push({role:'user',content:text});addTyping();
  try{
    const r=await fetch('https://api.anthropic.com/v1/messages',{method:'POST',headers:{'Content-Type':'application/json'},
      body:JSON.stringify({model:'claude-sonnet-4-20250514',max_tokens:300,system:cur.sys,messages:hist})});
    const d=await r.json();rmTyping();
    const rep=d.content?.[0]?.text||'...';
    hist.push({role:'assistant',content:rep});addMsg('ai',rep);
  }catch(e){rmTyping();addMsg('ai','Ошибка соединения. Попробуйте ещё раз.');}
  document.getElementById('send-btn').disabled=false;
}

function onKey(e){if(e.key==='Enter'&&!e.shiftKey){e.preventDefault();send();}}

async function endChat(){
  if(hist.length<2){alert('Проведите хотя бы несколько реплик для разбора.');return;}
  stopMic();go('results');
  document.getElementById('res-wrap').innerHTML=`<div class="res-loader"><div class="dots" style="justify-content:center"><span></span><span></span><span></span></div><br>Кристина анализирует ваш разговор...</div>`;
  const conv=hist.map(m=>`${m.role==='user'?'Руководитель':'Сотрудник'}: ${m.content}`).join('\n\n');
  const ep=`Ты — старший HRD-эксперт Кристина Плетнева. Проанализируй управленческий диалог из тренажёра по сценарию "${cur.title}".\n\nДИАЛОГ:\n${conv}\n\nОтвет строго в JSON (без markdown):\n{"score":<0-100>,"verdict":"<одна строка>","strengths":["<1>","<2>","<3>"],"gaps":["<1>","<2>","<3>"],"tip":"<конкретный совет на следующий раз>","competencies":{"Активное слушание":<0-10>,"Управление эмоциями":<0-10>,"Структура разговора":<0-10>,"Постановка вопросов":<0-10>,"Работа с сопротивлением":<0-10>,"Завершение диалога":<0-10>}}`;
  try{
    const r=await fetch('https://api.anthropic.com/v1/messages',{method:'POST',headers:{'Content-Type':'application/json'},
      body:JSON.stringify({model:'claude-sonnet-4-20250514',max_tokens:1000,messages:[{role:'user',content:ep}]})});
    const d=await r.json();
    let txt=(d.content?.[0]?.text||'{}').replace(/```json|```/g,'').trim();
    const ev=JSON.parse(txt);renderResults(ev);
    sess.push({title:cur.title,cat:cur.cat,score:ev.score,date:new Date().toLocaleDateString('ru')});
    try{localStorage.setItem('hrk_v4',JSON.stringify(sess))}catch(e){}
  }catch(e){document.getElementById('res-wrap').innerHTML='<div class="res-loader">Ошибка анализа. Попробуйте ещё раз.</div>';}
}

function renderResults(ev){
  const cr=Object.entries(ev.competencies||{}).map(([k,v])=>`<div class="cmp-row"><span class="cmp-l">${k}</span><div class="cmp-b"><div class="cmp-f" style="width:${v*10}%"></div></div><span class="cmp-n">${v}</span></div>`).join('');
  document.getElementById('res-wrap').innerHTML=`
    <div class="res-hero"><div class="res-n">${ev.score}</div><div class="res-l">из 100 · ${ev.verdict||''}</div></div>
    <div class="res-blk"><h3>Сильные стороны</h3><ul>${(ev.strengths||[]).map(s=>`<li>${s}</li>`).join('')}</ul></div>
    <div class="res-blk"><h3>Точки роста</h3><ul>${(ev.gaps||[]).map(g=>`<li>${g}</li>`).join('')}</ul></div>
    <div class="res-blk"><h3>Компетенции</h3>${cr}</div>
    <div class="res-blk"><h3>Совет Кристины</h3><p>${ev.tip||''}</p></div>
    <div class="res-acts">
      <button class="btn-o" onclick="startSc(cur)">Пройти ещё раз</button>
      <button class="btn-g" onclick="go('scenarios')">Другой сценарий</button>
    </div>`;
}

function renderProfile(){
  const t=sess.length;
  document.getElementById('p-s').textContent=t;
  document.getElementById('p-a').textContent=t?Math.round(sess.reduce((s,x)=>s+x.score,0)/t):'—';
  document.getElementById('p-b').textContent=t?Math.max(...sess.map(x=>x.score)):'—';
  const el=document.getElementById('hist');
  if(!t){el.innerHTML='<p style="font-size:13px;color:var(--ink2);padding:20px 0">Сессий пока нет.</p>';return;}
  el.innerHTML=sess.slice().reverse().map(s=>`<div class="hi"><div class="hi-sc">${s.score}</div><div><div style="font-family:var(--sf);font-size:14px;color:var(--ink)">${s.title}</div><div style="font-size:10px;color:var(--ink3);margin-top:3px">${s.cat} · ${s.date}</div></div></div>`).join('');
}

function mkCard(sc){
  const d=document.createElement('button');d.className='card';d.onclick=()=>startSc(sc);
  d.innerHTML=`<div class="cm"><span class="pill">${sc.cat}</span><span class="diff">${sc.diff}</span></div><h3>${sc.title}</h3><p>${sc.desc}</p><div class="cf"><span class="ct">${sc.time}</span><span class="cg">Начать →</span></div>`;
  return d;
}

function renderHomeGrid(){
  const g=document.getElementById('home-grid');
  SC.filter((_,i)=>i<4).forEach(sc=>g.appendChild(mkCard(sc)));
}

function renderGrid(f='Все'){
  const g=document.getElementById('full-grid');g.innerHTML='';
  const list=f==='Все'?SC:SC.filter(s=>s.cat===f);
  document.getElementById('sc-count').textContent=list.length+' сценариев';
  list.forEach(sc=>g.appendChild(mkCard(sc)));
}

function renderCats(){
  const cats=['Все',...new Set(SC.map(s=>s.cat))];
  const cf=document.getElementById('cats');
  cats.forEach(c=>{
    const b=document.createElement('button');b.className='cb'+(c==='Все'?' on':'');b.textContent=c;
    b.onclick=()=>{cf.querySelectorAll('.cb').forEach(x=>x.classList.remove('on'));b.classList.add('on');renderGrid(c);};
    cf.appendChild(b);
  });
}

renderHomeGrid();renderCats();renderGrid();
